const response = await fetch('https://shortmylink.in/api', {
  method: 'POST',
  headers: {
    'Authorization': 'YOUR_API_TOKEN'
  },
  body: JSON.stringify({
    url: 'https://your-long-url.com'
  })
});

// Response
const data = await response.json();
console.log(data.shortenedUrl);
